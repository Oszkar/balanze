# v0.3.0 — Tray Popover (hero) + Gauge Tray · Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers-extended-cc:subagent-driven-development (recommended) or superpowers-extended-cc:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Render the live `Snapshot` in a glanceable Tauri 2 tray popover (Grid + Cards, light/dark, badge-free Grid with hover provenance) and a color-shifting gauge tray icon, by booting the existing `watcher → state_coordinator → TauriSink` spine into the Tauri host.

**Architecture:** Reuse the actor spine unchanged. The Tauri host builds a tokio runtime, spawns the `StateCoordinator` with a production `TauriSink`, spawns the `Watcher` tasks against it, and manages the coordinator handle for IPC. `TauriSink::on_snapshot` emits `usage_updated` (the **raw `Snapshot`**) + repaints a runtime-rendered gauge tray icon (deduped); `on_degraded` emits `degraded_state`. The Svelte SPA reads the snapshot via `get_snapshot` + `usage_updated`, owns all popover presentation, and renders the two views.

**Tech Stack:** Tauri 2, Rust (tokio, `tauri::image::Image`), Svelte 5 runes, SvelteKit `adapter-static`, TypeScript, Vitest.

**Spec:** `docs/superpowers/specs/2026-06-03-v0.3.0-popover-design.md`. **Branch:** `docs/v0.3.0-popover-design` (continue here).

```ts
export type Tone = 'ok' | 'warn' | 'bad' | 'ink';
export interface PaceVerdict { ratio: number | null; text: string; tone: Tone; }

// Mirrors the wireframe paceVerdict + the CLI window::pace semantics.
export function paceVerdict(usedFraction: number, elapsedFraction: number): PaceVerdict {
  if (elapsedFraction < 0.04) return { ratio: null, text: 'too early', tone: 'ink' };
  const r = usedFraction / elapsedFraction;
  if (r >= 1.5) return { ratio: r, text: `${r.toFixed(1)}× faster than linear`, tone: 'bad' };
  if (r >= 1.12) return { ratio: r, text: `${r.toFixed(1)}× faster than linear`, tone: 'warn' };
  if (r <= 0.85) return { ratio: r, text: 'under linear pace', tone: 'ok' };
  return { ratio: r, text: 'on pace', tone: 'ok' };
}
```

```ts
// src/lib/presentation/format.ts
export const microUsdToDollars = (micro: number): string => `$${(micro / 1_000_000).toFixed(2)}`;

// "2h 41m", "3d 4h", "(passed)" — relative to now.
export function relativeReset(isoResetsAt: string, now: Date = new Date()): string {
  const ms = new Date(isoResetsAt).getTime() - now.getTime();
  if (ms <= 0) return '(passed)';
  const mins = Math.floor(ms / 60000);
  const d = Math.floor(mins / 1440), h = Math.floor((mins % 1440) / 60), m = mins % 60;
  if (d > 0) return `${d}d ${h}h`;
  if (h > 0) return `${h}h ${m}m`;
  return `${m}m`;
}

// "~3.2k/min", "~840/min", or "—" when null.
export function formatBurn(tokensPerMin: number | null): string {
  if (tokensPerMin == null) return '—';
  if (tokensPerMin >= 1000) return `~${(tokensPerMin / 1000).toFixed(1)}k/min`;
  return `~${Math.round(tokensPerMin)}/min`;
}
```

```ts
// src/lib/presentation/quota.ts
import type { Snapshot } from '../types/snapshot';
import type { Tone } from './pace';

export function quotaTone(pct: number): Tone {
  if (pct >= 90) return 'bad';
  if (pct >= 75) return 'warn';
  if (pct >= 50) return 'warn';
  return 'ok';
}

export interface QuotaWindow { pct: number; resetsAt: string; label: string; }
export interface AnthropicQuota {
  headline: QuotaWindow;        // 5-hour
  secondary: QuotaWindow | null; // 7-day
  source: 'statusline' | 'oauth';
  tone: Tone;
}

// Prefer the live statusline rate_limits; fall back to OAuth cadences.
export function anthropicQuota(s: Snapshot): AnthropicQuota | null {
  const sl = s.claude_statusline?.payload.rate_limits;
  if (sl?.five_hour) {
    const five = sl.five_hour;
    const seven = sl.seven_day;
    return {
      headline: { pct: five.used_percent, resetsAt: five.resets_at, label: '5h' },
      secondary: seven ? { pct: seven.used_percent, resetsAt: seven.resets_at, label: '7-day' } : null,
      source: 'statusline',
      tone: quotaTone(five.used_percent),
    };
  }
  const cad = s.claude_oauth?.cadences ?? [];
  const five = cad.find((c) => c.key === 'five_hour');
  if (!five) return null;
  const seven = cad.find((c) => c.key === 'seven_day');
  return {
    headline: { pct: five.utilization_percent, resetsAt: five.resets_at, label: '5h' },
    secondary: seven ? { pct: seven.utilization_percent, resetsAt: seven.resets_at, label: '7-day' } : null,
    source: 'oauth',
    tone: quotaTone(five.utilization_percent),
  };
}

// Codex window elapsed fraction (codex has no `pace` entry — compute it).
export function codexElapsedFraction(w: { resets_at: string; window_duration_minutes: number }, now = new Date()): number {
  const totalMs = w.window_duration_minutes * 60_000;
  const remainMs = new Date(w.resets_at).getTime() - now.getTime();
  if (totalMs <= 0) return 0;
  const elapsed = 1 - remainMs / totalMs;
  return Math.min(1, Math.max(0, elapsed));
}
```

```ts
// src/lib/stores/usage.svelte.ts
import type { UnlistenFn } from '@tauri-apps/api/event';
import { getSnapshot, refreshNow, onUsageUpdated, onDegraded } from '../ipc';
import type { Snapshot } from '../types/snapshot';

class UsageStore {
  snapshot = $state<Snapshot | null>(null);
  degraded = $state<Record<string, string>>({});
  loading = $state(true);
  lastError = $state<string | null>(null);
  #unlisten: UnlistenFn[] = [];

  async init() {
    try {
      this.snapshot = await getSnapshot();
    } catch (e) {
      this.lastError = String(e);
    } finally {
      this.loading = false;
    }
    this.#unlisten.push(await onUsageUpdated((s) => {
      this.snapshot = s;
      // A fresh snapshot clears degraded entries whose source now has data;
      // keep it simple: the degraded_state event is the source of truth.
    }));
    this.#unlisten.push(await onDegraded((d) => {
      this.degraded = { ...this.degraded, [d.source]: d.error };
    }));
  }

  async refresh() {
    try { await refreshNow(); } catch (e) { this.lastError = String(e); }
  }

  destroy() {
    for (const u of this.#unlisten) u();
    this.#unlisten = [];
  }
}

export const usage = new UsageStore();
```

```rust
        } else if util_percent >= 75.0 {

        Source::AnthropicApiCost => "anthropic_api_cost",
        Source::CodexQuota => "codex_quota",
        Source::OpenAiCosts => "openai_costs",

/// Worst-window utilization across all connected providers (0.0 when none).
pub(crate) fn worst_utilization(s: &Snapshot) -> f32 {
    let mut worst = 0.0_f32;
    if let Some(o) = &s.claude_oauth {
        for c in &o.cadences {
            worst = worst.max(c.utilization_percent);
        }
    }
    if let Some(sl) = &s.claude_statusline {
        if let Some(rl) = &sl.payload.rate_limits {
            if let Some(w) = &rl.five_hour {
                worst = worst.max(w.used_percent);
            }
            if let Some(w) = &rl.seven_day {
                worst = worst.max(w.used_percent);
            }
        }
    }
    if let Some(c) = &s.codex_quota {
        worst = worst.max(c.primary.used_percent as f32);
    }
    worst
}

impl TauriSink {
    pub(crate) fn new(app: AppHandle) -> Self {
        Self { app, last_painted: None }
    }

    /// Returns `Some((bucket,title))` to paint, or `None` if unchanged.
    fn paint_target(&self, s: &Snapshot, degraded: bool) -> (ColorBucket, String) {
        let bucket = if degraded {
            ColorBucket::Warn
        } else {
            ColorBucket::from_util(worst_utilization(s))
        };

    }
}

impl Sink for TauriSink {
    fn on_snapshot(&mut self, snapshot: &Snapshot) {
        if let Err(e) = self.app.emit("usage_updated", snapshot) {
            tracing::warn!("tauri_sink: emit usage_updated failed: {e}");
        }
        let target = self.paint_target(snapshot, /* degraded */ false);
        if self.last_painted.as_ref() != Some(&target) {
            tray_icon::paint(&self.app, target.0, &target.1);
            self.last_painted = Some(target);
        }
    }

    fn on_degraded(&mut self, source: Source, error: &str) {
        let payload = DegradedPayload { source: source_key(source).to_string(), error: error.to_string() };
        if let Err(e) = self.app.emit("degraded_state", payload) {
            tracing::warn!("tauri_sink: emit degraded_state failed: {e}");
        }
        // Flip the tray to the warning variant regardless of bucket.
        let title = self
            .last_painted
            .as_ref()
            .map(|(_, t)| t.clone())
            .unwrap_or_default();
        let target = (ColorBucket::Warn, title);
        if self.last_painted.as_ref() != Some(&target) {
            tray_icon::paint(&self.app, target.0, &target.1);
            self.last_painted = Some(target);
        }
    }
}
```

```rust
//! Runtime-rendered gauge tray icon. Hand-rolled RGBA raster (no extra dep):
//! a colored ring on a transparent background, color chosen by `ColorBucket`.
//! AGENTS.md §3.1: painting is deduped upstream in `tauri_sink`.

use tauri::image::Image;
use tauri::{AppHandle, Manager};

use crate::tauri_sink::ColorBucket;

/// Icon edge length in px. 32 matches the bundled `icons/32x32.png`.
const SIZE: u32 = 32;

/// (r,g,b) for each bucket — earthy status palette matching the UI tokens.
fn bucket_rgb(bucket: ColorBucket) -> (u8, u8, u8) {
    match bucket {
        ColorBucket::Green => (0x3f, 0x8f, 0x5f),
        ColorBucket::Yellow => (0xcf, 0x8a, 0x2a),
        ColorBucket::Orange => (0xd9, 0x6a, 0x2a),
        ColorBucket::Red => (0xc0, 0x49, 0x3a),
        ColorBucket::Warn => (0xc0, 0x49, 0x3a),
    }
}

/// Render a `size`×`size` RGBA ring in the bucket color on transparent bg.
pub(crate) fn render_gauge(bucket: ColorBucket, size: u32) -> Vec<u8> {
    let (r, g, b) = bucket_rgb(bucket);
    let mut buf = vec![0u8; (size * size * 4) as usize];
    let c = (size as f32 - 1.0) / 2.0;
    let outer = size as f32 / 2.0 - 1.0;
    let inner = outer - size as f32 * 0.22; // ring thickness ~22%
    for y in 0..size {
        for x in 0..size {
            let dx = x as f32 - c;
            let dy = y as f32 - c;
            let dist = (dx * dx + dy * dy).sqrt();
            // Anti-aliased ring membership.
            let a = if dist <= inner || dist >= outer {
                0.0
            } else {
                let edge = (1.0 - (dist - inner).min(outer - dist).min(1.0)).max(0.0);
                1.0 - edge * 0.0 + 1.0 - 1.0 // keep solid inside the band
            };
            let alpha = if dist > inner && dist < outer { 255 } else { (a * 255.0) as u8 };
            let i = ((y * size + x) * 4) as usize;
            buf[i] = r;
            buf[i + 1] = g;
            buf[i + 2] = b;
            buf[i + 3] = alpha;
        }
    }
    buf
}

/// Paint the tray icon + tooltip. Logs and no-ops on any Tauri error — a
/// failed repaint must never crash the coordinator task.
pub(crate) fn paint(app: &AppHandle, bucket: ColorBucket, title: &str) {
    let Some(tray) = app.tray_by_id("main") else {
        tracing::warn!("tray_icon: tray 'main' not found; skipping paint");
        return;
    };
    let rgba = render_gauge(bucket, SIZE);
    let img = Image::new_owned(rgba, SIZE, SIZE);
    if let Err(e) = tray.set_icon(Some(img)) {
        tracing::warn!("tray_icon: set_icon failed: {e}");
    }
    let _ = tray.set_tooltip(Some(if title.is_empty() { "Balanze" } else { title }));
    // macOS shows a title next to the icon; Windows ignores it.
    #[cfg(target_os = "macos")]
    {
        let _ = tray.set_title(Some(title));
    }
}
```

```rust
    let quit = MenuItemBuilder::with_id("quit", "Quit").build(app)?;

/// Boot the live data spine: coordinator + TauriSink + watcher, supervised.
/// Mirrors `balanze_cli::watch_cmd::run_with_sink` (AGENTS.md §4 #4/#7).
fn boot_backend(app: &App, rt: &tokio::runtime::Handle) {
    let _enter = rt.enter(); // so the coordinator/watcher `tokio::spawn`s land on `rt`

    let settings = settings::load().unwrap_or_else(|e| {
        tracing::warn!("settings load failed ({e}); using defaults");
        settings::Settings::default()
    });

    let sink = TauriSink::new(app.handle().clone());
    let (handle, coord_join) = state_coordinator::spawn(sink);
    app.manage(handle.clone());

    let watcher_handles = watcher::Watcher::spawn(handle, &settings);

    // Supervisor: log unexpected exits (a clean Ok(Ok(())) is NOT fatal).
    rt.spawn(async move {
        for (label, h) in watcher_handles {
            tokio::spawn(async move {
                match h.await {
                    Ok(Ok(())) => tracing::debug!("watcher/{label}: exited Ok(())"),
                    Ok(Err(e)) => tracing::error!("watcher/{label}: returned error: {e}"),
                    Err(je) => tracing::error!("watcher/{label}: panicked/aborted: {je}"),
                }
            });
        }
        match coord_join.await {
            Ok(()) => tracing::error!("state_coordinator task exited"),
            Err(je) => tracing::error!("state_coordinator panicked/aborted: {je}"),
        }
    });
}

pub fn run() {

        .with_env_filter(tracing_subscriber::EnvFilter::from_default_env())
        .init();

    // Dedicated multi-thread runtime (mirrors watch_cmd.rs). Managed by the app
    // so its worker threads live for the process lifetime.
    let rt = tokio::runtime::Builder::new_multi_thread()
        .enable_all()
        .build()
        .expect("build tokio runtime");
    let rt_handle = rt.handle().clone();

    let mut builder = tauri::Builder::default();
    #[cfg(any(target_os = "windows", target_os = "macos"))]
    {
        builder = builder.plugin(tauri_plugin_single_instance::init(|app, _args, _cwd| {
            position_and_show(app);
        }));
    }

    builder
        .plugin(tauri_plugin_opener::init())
        .manage(rt) // keep the runtime alive for the app's lifetime
        .invoke_handler(tauri::generate_handler![
            commands::get_snapshot,
            commands::refresh_now
        ])
        .on_window_event(|window, event| {
            // Hide-on-blur makes the window behave like a menubar popover.
            if let WindowEvent::Focused(false) = event {
                let _ = window.hide();
            }
        })
        .setup(move |app| {
            setup_tray(app)?;
            boot_backend(app, &rt_handle);
            Ok(())
        })
        .run(tauri::generate_context!())
        .expect("error while running tauri application");
}
```
